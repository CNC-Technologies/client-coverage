    index,
    username,
    objectContain = false,

    media: Media
    index: number
    username: string
    objectContain?: boolean
}) {
    const videoRef = useRef<HTMLVideoElement | null>(null)
 
    useEffect(() => {
        // Initialize Video.js
        if (videoRef.current) {
            const player = videojs(videoRef.current)
            // Clean up on component unmount
            return () => {
                player.dispose()
            }
        }
    }, [])
    return (
        <div className='h-full w-full'>
            {media?.MediaType?.includes('image') ? (
                <Image
                    height={100}
                    width={100}
                    unoptimized
                    priority
                    className={`w-full h-full ${
                        objectContain ? 'object-contain' : 'object-cover'
                    }  object-center  `}
                    src={media.URL}
                    alt={`${username}'s post on placio`}
                />
            ) : media?.MediaType?.includes('video') ? (
                <div className='relative h-full'>

                        ref={videoRef}

                </div>
            ) : (

}
