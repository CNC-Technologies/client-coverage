    MapMarkerOutlineIcon,

} from '@/constants/icons/icons'

import TicketPolicyVenue from '@/utils/FormValidators/EventCreation/ticketPolicyVenue'
import useLayoutConfig from '@/hooks/useLayoutConfig'
import { EventInterface } from '@/store/slices/events.slice'

const VenueLocation = dynamic(() => import('./VenueLocation'))

const InputField = dynamic(
    () => import('@/components/FormComponents/InputField'),
)

interface EventTicketAndVenueProps {
    event: EventInterface
    setEvent: React.Dispatch<React.SetStateAction<EventInterface>>
    nextStep
    setCoverImage: React.Dispatch<React.SetStateAction<any>>
}

export default function EventTicketAndVenue({
    event,
    setEvent,
    nextStep,
    setCoverImage,
}: EventTicketAndVenueProps) {
    const { errors, validate, validateCoverImage } = TicketPolicyVenue()

    const { configKey } = useLayoutConfig()

    const eventVenues = ['Virtual', 'In person', 'Hybrid']

    const [selectedFile, setSelectedFile] = useState<string | null>(null)
    // const [showTicketForm, setShowTicketForm] = useState(false)

    // Function to handle file upload
    const handleFileInputChange = (e: React.ChangeEvent<HTMLInputElement>) => {
        const file = e.target.files?.[0]
        setCoverImage(file)
        if (!file) return

        setSelectedFile(null)
        const reader = new FileReader()

        reader.onload = () => {
            const result = reader.result
            if (typeof result === 'string') {
                setSelectedFile(result)
                validateCoverImage(result)
            }
        }
        reader.readAsDataURL(file)
    }

    // Function to delete a single image on the image preview
    const handleDeleteButtonClick = () => {
        setSelectedFile(null)
    }

    // function to handle venue selection
    const handleVenueSelect = (name: string) => {
        setEvent({ ...event, venueType: name })
    }

    // function to handle input change
    const handleInputChange: React.ChangeEventHandler<HTMLInputElement> = (
        e,
    ) => {
        updateFormState(e?.target.name, e?.target.value, event, setEvent)
    }

    const handleSubmitWrapper = (e: Event) => {
        e.preventDefault()
        const venue = {
            location: event.location,
            coverImage: selectedFile,
            tags: event.tags,
        }
        const isValid = validate(venue, event.venueType)
        if (isValid) {
            // submit the form
            nextStep()
        }
    }

    // const deleteTicket = (indexToDelete: number) => {
    //     const filteredTickets = event?.tickets.filter(
    //         (_, index: number) => index !== indexToDelete,
    //     )
    //     setEvent({ ...event, tickets: filteredTickets })
    // }

    return (
        <div>
            <div className='px-4'>
                <div className='flex gap-2 items-center  my-4'>
                    <div className='eventCreationIcons'>
                        <MapMarkerOutlineIcon className='w-6 h-6' />
                    </div>

                    <h6 className='font-semibold text-xl max-mobile:text-lg '>
                        Set up your event venue
                    </h6>
                </div>

                <div className='mt-6 flex max-lgMobile:flex-col gap-2 lgMobile:gap-5'>
                    {eventVenues.map((item, index) => {
                        return (
                            <button
                                disabled={
                                    configKey !== 'business' &&
                                    item !== 'Virtual'
                                }
                                onClick={() => handleVenueSelect(item)}
                                role={item}
                                key={index}
                                className={`flex justify-center disabled:bg-grey/70 disabled:border-grey disabled:cursor-not-allowed w-full max-mobile:text-sm items-center rounded md text-center px-5 py-2 ${
                                    event.venueType === item
                                        ? 'orange-fill cursor-pointer'
                                        : 'orange-outline '
                                }  `}
                                // class for a normal user to disable other options
                                // { border border-primary/30 text-primary/30 cursor-not-allowed}
                            >
                                {item}
                            </button>
                        )
                    })}
                </div>
                {configKey !== 'business' && (
                    <p className='text-dark-charcoal/70 dark:text-white/70 max-lgMobile:text-sm mt-2'>
                        Note that Onsite venues and Hybrid venues are only
                        available for business accounts.
                    </p>
                )}

                {event.venueType !== 'In person' && (
                    <div className='mt-6'>
                        {event.venueType === 'Hybrid' && (
                            <div className='text-lg text-center  font-medium mb-3'>
                                For Virtual event attendees
                            </div>
                        )}
                        <InputField
                            data-testid='External event link'
                            name='url'
                            label={'External event link'}
                            value={event.url}
                            onChange={handleInputChange}
                        />
                    </div>
                )}

                {event.venueType !== 'Virtual' && (
                    <VenueLocation
                        errors={errors}
                        ticketAndVenueForm={event}
                        setEvent={setEvent}
                        handleInputChange={handleInputChange}
                    />
                )}
            </div>

            <hr className='my-8 border-darkGrey' />

            {/* Cover Image upload section */}
            <div className='px-4'>
                <div className='flex gap-2 items-center  my-4'>
                    <div className='eventCreationIcons'>
                        <MapMarkerOutlineIcon className='w-6 h-6' />
                    </div>

                    <h6 className='font-semibold text-xl max-mobile:text-lg '>
                        Others
                    </h6>
                </div>

                <div>
                    <div className=''>
                        <span className='text-lg font-medium'>Cover image</span>
                        <p className='font-light text-sm max-mobile:text-xs text-dark-charcoal/70 dark:text-white/70 '>
                            (Upload a copy of your cover image)
                        </p>
                    </div>

                    <ImageSelectItem
                        selectedFile={selectedFile}
                        handleDeleteButtonClick={handleDeleteButtonClick}
                        handleFileInputChange={handleFileInputChange}
                        error={errors.coverImage as string}
                    />
                </div>

                <hr className='my-8 border-darkGrey' />

                <div className='mt-6'>
                    <InputField
                        data-testid='Tags'
                        name='tags'
                        label={'Tags '}
                        value={event.tags}
                        onChange={handleInputChange}
                        placeholder={'tech, design, cool...'}
                        error={errors.tags}
                    />
                    <span className='mt-3 text-dark-charcoal/70 dark:text-white/70 text-sm'>
                        (Improve discovery of your event by adding relevant
                        keywords to the event subject)
                    </span>
                </div>
            </div>

            <hr className='my-8 border-darkGrey' />

            <hr className='my-8 border-darkGrey' />

            <div className='flex w-full flex-col mt-4 px-4 gap-8'>
                {/* <OutlineButton
                    name={'Save and finish later'}
                    className='orange-outline'
                >
                    
                </OutlineButton> */}

                <Button
                    // disabled={!isFormFilled()}
                    onClick={handleSubmitWrapper}
                    name={'Save and Continue'}
                    className='orange-fill'
                />
            </div>
        </div>
    )
}
