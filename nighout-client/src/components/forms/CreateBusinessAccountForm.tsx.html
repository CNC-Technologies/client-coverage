import Image from 'next/image'

import { ChangeEvent, useContext, useRef, useState } from 'react'

import { useRouter } from 'next/router'

import { useDispatch } from 'react-redux'

import InputField from '@/components/FormComponents/InputField'
import Button, { IconButton } from '@/components/FormComponents/button'
import Avatar from '@/components/app/simple/Avatar'
import { ToastContext } from '@/components/context-containers/Toast.context'
import { defaultCoverPhoto, defaultPicture } from '@/constants/app/defaults'
import { businessRoute } from '@/constants/app/routes'
import { CloseOutlineIcon, EditIcon } from '@/constants/icons/icons'
// import UploadForm from '@/hooks/useFileUpload'
import useSecureRequest from '@/hooks/useRequest'
import { imagesType } from '@/interfaces/settings.interface'
import { setActiveBusiness } from '@/store/slices/activeBusiness.slice'
import useUploadMedia from '@/hooks/useMediaUpload'

interface BusinessInfoInterface {
    picture: any
    cover_image: any
    name: string
    username: string
    description: string
    website: string
}

export default function CreateBusinessAccountForm({ handleModal }) {
    const router = useRouter()

    const dispatch = useDispatch()

    const createBusinessReq = useSecureRequest(businessRoute)

    // const { handleUpload, handleDelete } = UploadForm()

    const { showToast } = useContext(ToastContext)

    const pictureInputRef = useRef<HTMLInputElement>(null)
    const coverInputRef = useRef<HTMLInputElement>(null)

    const [formData, setFormData] = useState<BusinessInfoInterface>({
        picture: defaultPicture,
        cover_image: defaultCoverPhoto,
        name: '',
        username: '',
        description: '',
        website: '',
    })
    const [creatingBusiness, setCreatingBusiness] = useState(false)

    const { uploadMedia } = useUploadMedia()

    // eslint-disable-next-line
    const [images, setImages] = useState<imagesType>({
        picture: null,
        cover_image: null,
    })
    const [prevImages, setPrevImages] = useState<imagesType>({
        picture: '',
        cover_image: '',
    })

    const triggerPictureFileSelector = () => pictureInputRef?.current?.click()
    const triggerCoverFileSelector = () => coverInputRef?.current?.click()

    async function selectImage(key: string, e: ChangeEvent<HTMLInputElement>) {
        const file = e?.target?.files

        if (!file) showToast('error', 'No file was selected.')

        const url = URL.createObjectURL(file![0])

        if (key.includes('cover')) {
            if (!prevImages?.cover_image)
                setPrevImages!((prev) => ({
                    ...prev,
                    cover_image: formData.cover_image,
                }))
            setImages!((prev: imagesType) => ({ ...prev, cover_image: file }))
        } else {
            if (!prevImages?.picture)
                setPrevImages!((prev) => ({
                    ...prev,
                    picture: formData.picture,
                }))
            setImages!((prev: imagesType) => ({ ...prev, picture: file }))
        }
        setFormDataHandler(key, url)
    }

    function setFormDataHandler(fields: string, value: any) {
        if (fields === 'username') {
            value = value
                .trim()
                .toLowerCase()
                .replace(/[^a-zA-Z0-9_]/g, '')
        }
        setFormData((prevState) => {
            if (!fields && value) return value

            const newState = { ...prevState } // Create a shallow copy of the previous state
            // Update the specified field with the new value
            const fieldPath = fields.split('.')
            let current = newState
            for (let i = 0; i < fieldPath.length - 1; i++) {
                current = current[fieldPath[i]]
            }
            current[fieldPath[fieldPath.length - 1]] = value

            return current // Return the updated state
        })
    }

    function handleDeselect(key: string) {
        if (key.includes('cover')) {
            if (coverInputRef?.current?.value) coverInputRef.current.value = ''
        } else {
            if (pictureInputRef?.current?.value)
                pictureInputRef.current.value = ''
        }
        const prev = key.includes('cover')
            ? prevImages!.cover_image
            : prevImages!.picture
        setFormDataHandler(key, prev)
        if (key.includes('cover')) {
            setPrevImages!((prev) => ({ ...prev, cover_image: '' }))
        } else {
            setPrevImages!((prev) => ({ ...prev, picture: '' }))
        }
    }

    async function handleCreateBusiness() {
        setCreatingBusiness(true)
        let formDataCopy = { ...formData }
        try {
            // * validation (should only validate name)
            if (!formData.name)
                return showToast('error', 'Business name required!')

            // * upload images
            if (images.cover_image) {
                const res = await uploadMedia(images.cover_image)
                const img = res[0]?.URL
                formDataCopy = { ...formDataCopy, cover_image: img }
            }
            if (images.picture) {
                const res = await uploadMedia(images.picture)
                const img = res[0]?.URL
                formDataCopy = { ...formDataCopy, picture: img }
            }

            // * upload post
            const res = await createBusinessReq.post({ data: formDataCopy })
            if (res?.success) {
                handleModal(false)
                dispatch(setActiveBusiness(res.data))

                // For the tour from the user section
                const action = (router.query?.action as string) || ''

                router.push(`/business${action ? `?action=${action}` : ''}`)
            } else {
                throw 'Error saving business information'
            }
        } catch ({ error }) {
            // * Delete files if files were uploaded
            // if (formDataCopy.picture !== defaultPicture)
            //     handleDelete(formDataCopy.picture)
            // if (formDataCopy.cover_image !== defaultCoverPhoto)
            //     handleDelete(formDataCopy.cover_image)

            showToast(
                'error',
                `There was an error while creating your business account! ${error}`,
            )
        } finally {
            setCreatingBusiness(false)
        }
    }

    return (
        <div className='space-y-3 hideScrollbar'>
            <input
                ref={pictureInputRef}
                type='file'
                className='hidden'
                accept='image/*'
                onChange={(e: ChangeEvent<HTMLInputElement>) => {
                    selectImage('picture', e)
                }}
            />
            <input
                ref={coverInputRef}
                type='file'
                className='hidden'
                accept='image/*'
                onChange={(e: ChangeEvent<HTMLInputElement>) => {
                    selectImage('cover_image', e)
                }}
            />
            <div className='relative mt-0'>
                <div className='h-28 overflow-hidden relative'>
                    <Image
                        src={formData.cover_image}
                        height={100}
                        unoptimized
                        width={100}
                        onError={() => {
                            setFormDataHandler('cover_image', defaultCoverPhoto)
                        }}
                        alt='User cover image'
                        className='object-cover h-full w-full my-auto rounded-lg'
                        priority
                    />
                    <div className='absolute bottom-2 right-2'>
                        {prevImages!.cover_image ? (
                            <IconButton
                                variant='solid'
                                onClick={() => handleDeselect('cover_image')}
                                className='primary-fill'
                            >
                                <CloseOutlineIcon size={24} />
                            </IconButton>
                        ) : (
                            <IconButton
                                variant='solid'
                                onClick={triggerCoverFileSelector}
                                className='primary-fill'
                            >
                                <EditIcon size={24} />
                            </IconButton>
                        )}
                    </div>
                </div>
                <div className='absolute -bottom-9 left-1/2 -translate-x-1/2'>
                    <div className='rounded-full relative z-20'>
                        <Avatar size='xl' img={formData.picture} />
                        <div className='absolute bottom-0 right-0'>
                            {prevImages!.picture ? (
                                <IconButton
                                    variant='solid'
                                    onClick={() => handleDeselect('picture')}
                                    className='primary-fill'
                                >
                                    <CloseOutlineIcon size={24} />
                                </IconButton>
                            ) : (
                                <IconButton
                                    variant='solid'
                                    onClick={triggerPictureFileSelector}
                                    className='primary-fill'
                                >
                                    <EditIcon size={24} />
                                </IconButton>
                            )}
                        </div>
                    </div>
                </div>
            </div>
            <div className='pt-10 space-y-3'>
                <InputField
                    label='Business Name'
                    value={formData.name}
                    onChange={(e) => setFormDataHandler('name', e.target.value)}
                />
                <InputField
                    label='Business Username'
                    subDescription=' (Only accepts letters, numbers and underscores.)'
                    value={formData.username}
                    onChange={(e) =>
                        setFormDataHandler('username', e.target.value)
                    }
                />
                <InputField
                    label='Description'
                    type='textarea'
                    value={formData.description}
                    onChange={(e) =>
                        setFormDataHandler('description', e.target.value)
                    }
                />
                <InputField
                    label='Website'
                    type='url'
                    value={formData.website}
                    onChange={(e) =>
                        setFormDataHandler('website', e.target.value)
                    }
                />
                <div className='flex justify-center'>
                    <Button
                        className='w-fit'
                        onClick={handleCreateBusiness}
                        loading={creatingBusiness}
                    >
                        Create
                    </Button>
                </div>
            </div>
        </div>
    )
}
