import React, {
    createContext,
    useContext,
    useEffect,
    useState,
    ReactNode,
} from 'react'

import { MenuItemProps } from '@/interfaces/smartMenu.interface'

interface OrdersContextType {
    orders: { [restaurantId: string]: MenuItemProps[] }
    // eslint-disable-next-line no-unused-vars
    addOrder: (restaurantId: string, order: MenuItemProps) => void
    // eslint-disable-next-line no-unused-vars
    removeOrder: (restaurantId: string, orderId: string) => void
}

const OrdersContext = createContext<OrdersContextType | undefined>(undefined)

export const useOrdersCart = (): OrdersContextType => {
    const context = useContext(OrdersContext)
    if (!context) {
        throw new Error('useOrdersCart must be used within a OrdersProvider')
    }
    return context
}

export const OrdersProvider = ({ children }: { children: ReactNode }) => {
    const [orders, setOrders] = useState<{
        [restaurantId: string]: MenuItemProps[]
    }>(() => {
        // Retrieve stored orders from localStorage or initial state
        const storedOrders =
            typeof window !== 'undefined' &&
            localStorage.getItem('restaurant_orders')
        return storedOrders ? JSON.parse(storedOrders) : {}
    })

    useEffect(() => {
        // Update localStorage whenever orders change
        localStorage.setItem('restaurant_orders', JSON.stringify(orders))
    }, [orders])

    const addOrder = (restaurantId: string, order: MenuItemProps): void => {
        // Add a new order to the specified restaurant
        setOrders((prevOrders) => ({
            ...prevOrders,
            [restaurantId]: [...(prevOrders[restaurantId] || []), order],
        }))
    }

    const removeOrder = (restaurantId: string, orderId: string): void => {
        // Remove an order from the specified restaurant
        setOrders((prevOrders) => ({
            ...prevOrders,
            [restaurantId]: prevOrders[restaurantId].filter(
                (order) => order.id !== orderId,
            ),
        }))
    }

    const ordersContextValue = {
        orders,
        addOrder,
        removeOrder,
    }

    return (
        <OrdersContext.Provider value={ordersContextValue}>
            {children}
        </OrdersContext.Provider>
    )
}

export default OrdersProvider
