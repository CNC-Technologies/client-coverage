import { useSelector } from 'react-redux'
import React, { useContext, useState } from 'react'
import { useRouter } from 'next/router'

import InputField from '@/components/FormComponents/InputField'
import Button, { IconButton } from '@/components/FormComponents/button'
import useOrders from '@/hooks/business/places/smart-menu/useOrder'
import { RootState } from '@/store'
import AuthModalContext from '@/components/context-containers/AuthModal.context'
import { PlaceProps } from '@/interfaces/places/place.interface'
import { useBusinessWebsiteCart } from '@/components/context-containers/BusinessWebsiteCart'

export default function BusinessOrderItem({
    // eslint-disable-next-line no-unused-vars
    placeOrder,
    businessCartItems,
    businessData,
}) {
    const router = useRouter()

    const businessName = (router.query?.businessUsername as string) || ''

    const user = useSelector((state: RootState) => state.user.user)

    // eslint-disable-next-line no-unused-vars
    const { updateCartItem, carts, clearCart } = useBusinessWebsiteCart()

    const calculateTotal = (): string => {
        const items = (Object.values(businessCartItems) as any) || [] //! fix type
        const flattenedItems = items.flat() // Flatten the array of arrays

        return flattenedItems.reduce(
            (acc, item) => acc + item.price * (item.quantity || 1),
            0,
        )
    }

    // eslint-disable-next-line no-unused-vars
    const [additionalNote, setAdditionalNote] = useState('')

    const { showAuthModal } = useContext(AuthModalContext)

    const orderOption = ['Dine In', 'Takeaway']
    const [activeOption, setActiveOption] = useState('Dine In')

    const { createOrder } = useOrders()

    const getPlaceInBusiness = (placeId: string) => {
        const businessPlaces = businessData?.edges?.business?.edges
            ?.places as PlaceProps[]
        const place = businessPlaces.find((item) => item.id === placeId)
        return place ? place : null
    }

    const handleOrder = () => {
        if (!user) {
            showAuthModal()
            return
        }

        //     data,
        //     tableId: tableId,
        // }

        // createOrder.mutateAsync(finalData).then((res) => {
        //     // console.log(res)

        //     placeOrder()
        //     clearCart(placeId)
        //     addOrder(placeId, res)
        // })
    }

    return (
        <div className='mt-5'>
            <div className='flex justify-center space-x-3'>
                {orderOption.map((item, idx) => {
                    return (
                        <Button
                            key={idx}
                            className={`${
                                activeOption === item
                                    ? 'dark:hover:bg-orange'
                                    : 'bg-transparent primary-outline dark:bg-transparent dark:border-orange dark:text-white dark:hover:border-orange dark:hover:text-white dark:hover:bg-transparent hover:bg-transparent'
                            } `}
                            onClick={() => setActiveOption(item)}
                        >
                            {item}
                        </Button>
                    )
                })}
            </div>

            <hr className='border-black/40 border-dashed my-5' />

            <div className='grid gap-2'>
                {Object.entries(businessCartItems).map(
                    ([key, orders]: [string, any]) => {
                        // console.log(orders)
                        return (
                            <React.Fragment key={key}>
                                <div className='text-lg md:text-xl text-center capitalize dark:text-white text-black font-semibold'>
                                    {getPlaceInBusiness(key)?.name}
                                </div>

                                {orders?.map((order, idx) => {
                                    return (
                                        <div
                                            key={idx}
                                            className='bg-white dark:bg-black/10 p-2 rounded-xl h-fit'
                                        >
                                            <div className='flex flex-nowrap space-x-3'>
                                                <div className='w-full '>
                                                    <h5 className='font-medium'>
                                                        {order?.quantity} X{' '}
                                                        {order?.name}
                                                    </h5>
                                                </div>
                                                <div className=' flex items-center space-x-5 '>
                                                    <IconButton
                                                        onClick={() =>

                                                                businessName,

                                                        }

                    },
                )}
            </div>

                <InputField
                    type='textarea'
                    className='bg-[whitesmoke]'
                    name='note'
                    rows={3}
                    placeholder='Additional notes  or allergies'
                    onChange={(e) => setAdditionalNote(e.target.value)}
                />
            </div>

            <hr className='border-black/40 border-dashed my-5' />

            <div className='grid gap-1'>
                <div className='flex items-center justify-between'>

                    </div>
                </div>
            </div>

            <div className='flex my-3 justify-center'>
                <Button loading={createOrder.isLoading} onClick={handleOrder}>
                    Place Order
                </Button>
            </div>
        </div>
    )
}
