import Image from 'next/image'

import { IconButton } from '@/components/FormComponents/button'
import MenuList from '@/components/app/simple/MenuList'
import { ThreeDotsIcon } from '@/constants/icons/icons'

export default function MediaItem({
    item,
    type = 'image',
}: {
    item: any
    type: 'image' | 'video'
}) {
    const optionsItems = [
        {
            label: 'Delete',
            onClick: () => null,
        },
    ]

    if (type === 'image') {
        optionsItems.push({
            label: 'Set as Cover Photo',
            onClick: () => null,
        })
    }

    return (
        <div className='h-[150px] w-full relative bg-black rounded-xl overflow-hidden'>
            <span className='absolute top-2 right-2'>
                <MenuList
                    leftSided
                    menuButton={
                        <IconButton className='bg-black/10'>
                            <ThreeDotsIcon color='white' />
                        </IconButton>
                    }
                    menuItems={optionsItems}
                />
            </span>
            {item.MediaType === 'image' && (
                <Image
                    src={item.URL}
                    unoptimized
                    alt='Place media'
                    width={100}
                    height={100}
                    className='object-cover h-full w-full'
                />
            )}

            {item.MediaType === 'video' && (
                <video
                    controls={true}
                    preload='metadata'
                    className={`object-cover object-center h-full w-full`}
                >
                    <source
                        src={item.URL}
                        type={`video/${item.URL.split('.').reverse()[0]}`}
                    />
                </video>
            )}
        </div>
    )
}
