import { IconButton } from '@/components/FormComponents/button'
import FollowButton from '@/components/app/simple/FollowButton'
import LikeItem from '@/components/app/simple/LIkeItem'
import OverlappingAvatars from '@/components/app/simple/OverlappingAvatars'
import ShareButton from '@/components/app/simple/ShareButton'
import RatingStars from '@/components/ratingStars'
import { ForwardOutlineIcon, MapMarkerSolidIcon } from '@/constants/icons/icons'
import usePlaces from '@/hooks/app/place/usePlaces'

export default function PlaceDetailsHeading({ place }) {
    const {
        removePlaceLike,
        createPlaceLike,
        handleFollowPlace,
        handleUnfollowPlace,
    } = usePlaces()

    const baseUrl =
        typeof window !== 'undefined' &&
        `${window.location.protocol}//${window.location.hostname}`

    return (
        <div className='flex justify-between space-x-3 '>
            <div>
                <h2 className='text-2xl font-semibold capitalize'>
                    {place?.name}
                </h2>
                <RatingStars rating={4} />
                <div className='flex space-x-2 items-center'>
                    <OverlappingAvatars />
                    <p>Reviews</p>
                </div>
                <div className='flex space-x-2'>
                    <MapMarkerSolidIcon className='text-xl text-orange' />
                    <p className='font-medium'>{place?.location}</p>
                </div>
            </div>

            <div className='flex max-mobile:flex-col max-mobile:space-y-2 mobile:space-x-3 items-center h-fit'>
                <div className='flex items-center space-x-2 mobile:space-x-3'>
                    <div>
                        <LikeItem
                            likedByMe={
                                place?.likedByCurrentUser
                                    ? place?.likedByCurrentUser
                                    : false
                            }
                            likeAction={() => createPlaceLike.mutate(place?.id)}
                            unlikeAction={() =>
                                removePlaceLike.mutate(place?.id)
                            }
                        />
                    </div>
                    <ShareButton
                        content={{
                            url: `${baseUrl}/places/${place?.id}`,
                            text: `Visit ${place?.name} on Placio.`,
                        }}
                        type='place'
                        dataItem={place}
                    >
                        <IconButton className=' flex gap-1 items-center rounded-full justify-center text-center primary-outline'>
                            <ForwardOutlineIcon className='h-6 w-6' />
                        </IconButton>
                    </ShareButton>
                </div>
                <FollowButton
                    followedByMe={
                        place?.followedByCurrentUser
                            ? place?.followedByCurrentUser
                            : false
                    }
                    followAction={() => handleFollowPlace(place?.id)}
                    unFollowAction={() => handleUnfollowPlace(place?.id)}
                />
            </div>
        </div>
    )
}
