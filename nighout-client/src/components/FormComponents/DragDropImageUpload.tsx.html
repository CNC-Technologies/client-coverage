import Image from 'next/image'
 
import React, { ChangeEvent, DragEvent, useRef, useState } from 'react'
import { Navigation } from 'swiper'
import { Swiper, SwiperSlide } from 'swiper/react'
 
import { titleCase } from 'title-case'
 
import { IconButton } from '@/components/FormComponents/button'
import {
    ChevronLeftIcon,
    ChevronRightIcon,
    CloseOutlineIcon,
} from '@/constants/icons/icons'
import {
    DragDropImageUploadProps,
    FilePreviewItemProps,
    Photo,
} from '@/interfaces/dragDropImageUpload.interface'
 
const DragDropImageUpload: React.FC<DragDropImageUploadProps> = ({
    accept = 'image',
    videoControl = false,
    maxSelectable = 0,
    handleMediaChange,
    selectedPhotos,
    setSelectedPhotos,
    imageForUpload,
    setImageForUpload,
    optionalAltText = '',
    setImagesForDelete = () => null,
    allowMultiple = true,
}) => {
    const [isDragOver, setIsDragOver] = useState(false)
 
    const fileInputRef = useRef<HTMLInputElement>(null)
 
    const handleDragOver = (event: DragEvent<HTMLDivElement>) => {
        event.preventDefault()
        setIsDragOver(true)
    }
 
    const handleDragLeave = () => {
        setIsDragOver(false)
    }
 
    const handleFileChange = (files: FileList) => {
        handleMediaChange(files)
        const newSelectedPhotos: Photo[] = Array.from(files).map((file) => ({
            name: file.name,
            MediaType: file.type,
            URL: URL.createObjectURL(file),
        }))
 
        setSelectedPhotos((prevSelectedPhotos) => {
            const totalSelectedPhotos = [
                ...prevSelectedPhotos,
                ...newSelectedPhotos,
            ]
            return maxSelectable > 0
                ? totalSelectedPhotos.slice(0, maxSelectable)
                : totalSelectedPhotos
        })
    }
 
    const handleDrop = (event: DragEvent<HTMLDivElement>) => {
        event.preventDefault()
        setIsDragOver(false)

        const files = event.dataTransfer.files as FileList

        // This should only work if there is no specified maximum number of photos and when the number of images in the selectedPhotos array is less than the number specified as maximum images to be selected
        const shouldHandleFileChange =
            (selectedPhotos.length < maxSelectable && maxSelectable > 0) ||
            maxSelectable === 0

        if (shouldHandleFileChange) {
            handleFileChange(files)
        }
    }
 
    const handleInputChange = (event: ChangeEvent<HTMLInputElement>) => {
        const files = event.target.files as FileList
        handleFileChange(files)
    }
 
    // Function to trigger the input field when you click on the grey div
    const handleBoxClick = () => {
        const shouldHandleFileChange =
            (selectedPhotos.length < maxSelectable && maxSelectable > 0) ||
            maxSelectable === 0

        if (fileInputRef.current && shouldHandleFileChange) {
            fileInputRef.current.click()
        }
    }
 
    const removeSinglePreviewImage = (image: Photo) => {
        // Delete from the preview state
        const filteredPreviewImages = selectedPhotos.filter(
            (photo) => photo.URL !== image?.URL,
        )
 
        // For deleting Images from cloudinary (This only happens when you are trying to edit images)
        setImagesForDelete(image)
 
        setSelectedPhotos(filteredPreviewImages)
 
        // Delete from imageForUpload array
        const filteredImagesForUpload: File[] = []
        for (const key in imageForUpload) {
            if (Object.prototype.hasOwnProperty.call(imageForUpload, key)) {
                const file = imageForUpload[key]
                if (file.name !== image?.name) {
                    filteredImagesForUpload.push(file)
                }
            }
        }

        setImageForUpload(filteredImagesForUpload)
    }
 
    return (
        <div>
            <div>
                {selectedPhotos.length > 0 ? (
                    <div>
                        {maxSelectable > 0 && (
                            <p
                                className={`${
                                    selectedPhotos.length >= maxSelectable
                                        ? 'error-text'
                                        : ''
                                }`}
                            >
                                {selectedPhotos.length} out of {maxSelectable}{' '}
                                selected
                            </p>
                        )}
                        <div className=' grid grid-cols-1  gap-4 relative'>
                            <div className='w-full h-56'>
                                <Swiper
                                    initialSlide={selectedPhotos.length - 1}
                                    loop={true}
                                    navigation={{
                                        prevEl: '.custom-prev-button',
                                        nextEl: '.custom-next-button',
                                    }}
                                    modules={[Navigation]}
                                    className='h-full w-full'
                                >
                                    {selectedPhotos.map(
                                        (file: Photo, idx: number) => (
                                            <SwiperSlide key={idx}>
                                                <FilePreviewItem
                                                    file={file}
                                                    videoControl={videoControl}
                                                    removeSinglePreviewImage={
                                                        removeSinglePreviewImage
                                                    }
                                                    optionalAltText={
                                                        optionalAltText
                                                    }
                                                />
                                            </SwiperSlide>
                                        ),
                                    )}
 
                                    <div>
                                        <button className='custom-prev-button absolute z-10 transform -translate-y-1/2 rounded-full shadow-sm top-1/2 left-2 white-fill dark:bg-grey'>
                                            <ChevronLeftIcon className='w-6 h-6' />
                                        </button>
                                        <button className='custom-next-button absolute z-10 transform -translate-y-1/2 rounded-full shadow-sm top-1/2 right-2 white-fill dark:bg-grey'>
                                            <ChevronRightIcon className='w-6 h-6' />
                                        </button>
                                    </div>
                                </Swiper>
                            </div>
                        </div>
                    </div>
                ) : null}
            </div>
 
            <div
                className={`h-40 mt-5 cursor-pointer bg-white dark:bg-black dark:text-gray-300 flex items-center justify-center border border-dashed dark:border-gray-800 border-gray-300 rounded-lg ${
                    isDragOver ? 'border-blue-500' : ''
                }`}
                onDragOver={handleDragOver}
                onDragLeave={handleDragLeave}
                onDrop={handleDrop}
                onClick={handleBoxClick}
            >
                <input
                    name='images'
                    data-testid='file-input'
                    type='file'
                    accept={
                        accept === 'all' ? 'image/*, video/*' : `${accept}/*`
                    }
                    multiple={allowMultiple}
                    onChange={handleInputChange}
                    ref={fileInputRef}
                    className='hidden'
                />
                <div>
                    <div className='text-center mb-2'>
                        Click or Drag and Drop{' '}
                        {accept === 'all'
                            ? 'Images and Video'
                            : titleCase(accept)}{' '}
                        Here
                    </div>
                </div>
            </div>
        </div>
    )
}
 
function FilePreviewItem({
    file,
    removeSinglePreviewImage,
    videoControl,
    optionalAltText,
}: FilePreviewItemProps) {
    // * To preview videos
    if (
        (file.name as string)?.match(/\.(mp4|3gp|mov|wmv|mkv|webm)$/i) ||
        file.MediaType?.includes('video')
    )
        return (
            <div className='relative h-full w-full'>
                <video
                    controls={videoControl}
                    preload='metadata'
                    className={`object-cover object-center h-full w-full rounded`}
                >
                    <source src={file.URL} type='video/mp4' />
                </video>
                <div className='photo-name text-xs'>{file.name}</div>
                <IconButton
                    onClick={() => removeSinglePreviewImage(file)}
                    className='absolute top-2 right-2 bg-white'
                    data-testid='remove-button'
                >
                    <CloseOutlineIcon className='text-[red] dark:text-white' />
                </IconButton>
            </div>
        )
    if (
        (file.name as string)?.match(/\.(jpg|jpeg|png|gif|svg)$/i) ||
        file.MediaType?.includes('image')
    )
        return (
            <div className='relative'>
                <Image
                    unoptimized
                    src={file.URL}
                    alt={(file.name as string) || `${optionalAltText}`}
                    priority
                    quality={75}
                    width={100}
                    height={100}
                    className={`object-fill object-center h-full w-full `}
                />
                {/* <div className='photo-name'>{file.name}</div> */}
                <IconButton
                    onClick={() => removeSinglePreviewImage(file)}
                    className='absolute top-2 right-2 bg-white'
                    data-testid='remove-button'
                >
                    <CloseOutlineIcon color='red' />
                </IconButton>
            </div>
        )
    return null
}
 
export default DragDropImageUpload
