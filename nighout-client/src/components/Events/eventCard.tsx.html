import Image from 'next/image'
import Link from 'next/link'
 
import { useRouter } from 'next/router'
 
import { useContext } from 'react'
 
import { useUser } from '@auth0/nextjs-auth0/client'
import dynamic from 'next/dynamic'
 
import OverlappingAvatars from '@/components/app/simple/OverlappingAvatars'
 
import Truncate from '@/components/app/simple/Truncate'
 
import LikeItem from '@/components/app/simple/LIkeItem'
import { MailDialogContext } from '@/components/context-containers/MailDialog.context'
import {
    HorzMenuSolidIcon,
    MapMarkerOutlineIcon,
} from '@/constants/icons/icons'
import useLayoutConfig from '@/hooks/useLayoutConfig'
 
const MenuList = dynamic(import('@/components/app/simple/MenuList'))
 
export default function EventCard({ event }) {
    const { showMailDialog } = useContext(MailDialogContext)
    const { user } = useUser()
    const router = useRouter()
    const { configKey } = useLayoutConfig()
 
    // * Event details url based on a user being logged in and based on the configKey
    const detailsRoute = !user
        ? `/events/${event?.title}`
        : configKey === 'business'
        ? `/business/events/${event?.title}`
        : `/app/events/${event?.title}`
 
    const routeTo = (page: string) => router.push(`${detailsRoute}/${page}`)
    return (
        <div className='flex flex-col w-full rounded-xl bg-white'>
            <div className='relative h-[10rem] mobile:h-[12rem]'>
                <div className='absolute top-0 left-0 flex items-center justify-between w-full px-3 py-3 text-sm rounded-t-xl '>
                    <div className='flex items-center justify-center px-2 py-1 rounded-lg shadow-lg bg-white text-dark-charcoal dark:bg-darkGrey dark:text-dark-charcoal'>
                        <p className='font-semibold '>${event?.price}</p>
                    </div>
 
                    {configKey === 'business' && (
                        <MenuList
                            leftSided
                            menuButton={
                                <div>
                                    <div className='flex pointer-events-auto items-center justify-center px-2 py-1 rounded-lg shadow-lg bg-white text-dark-charcoal dark:bg-darkGrey dark:text-[rgb(25,_48,_60)]'>
                                        <p className='font-medium'>
                                            <HorzMenuSolidIcon />
                                        </p>
                                    </div>
                                </div>
                            }
                            menuItems={[
                                {
                                    label: 'View Feedback',
                                    onClick: () => routeTo('feedback'),
                                },
                                {
                                    label: 'View Channel',
                                    onClick: () => routeTo('channel'),
                                },
                                {
                                    label: 'Mail Channel',
                                    onClick: () => showMailDialog(),
                                },
                                {
                                    label: 'Delete Event',
                                    onClick: () => null,
                                },
                            ]}
                        />
                    )}
 
                    {configKey !== 'business' && <LikeItem />}
                </div>
                {/* <Link href={detailsRoute}> */}
                <Image
                    src={event?.imageSrc}
                    alt={'category'}
                    priority={true}
                    quality={75}
                    width={100}
                    height={100}
                    className='object-cover object-center w-full h-full z-10 rounded-xl cursor-pointer'
                    onClick={() => router.push(detailsRoute)}
                />
                {/* </Link> */}
                <div className='absolute bottom-0 left-0 pointer-events-none flex flex-wrap justify-between w-full px-3 py-2 '>
                    <div className='bg-white dark:bg-darkGrey pointer-events-auto text-dark-charcoal dark:text-dark-charcoal shadow-lg py-1 px-2 xmobile:px-4 text-sm rounded-lg flex flex-col justify-center items-center'>
                        <p className='font-semibold'>31</p>
                        <p className=''>Mar</p>
                    </div>
                </div>
            </div>
            {/* <Link href={detailsRoute}> */}
            <Link
                href={detailsRoute}
                className='w-full px-3 py-3 text-dark-charcoal h-fit rounded-b-xl cursor-pointer'
                // onClick={() => router.push(detailsRoute)}
            >
                <Truncate outerClass='font-medium'>{event?.title}</Truncate>
                {/* <p className='font-medium truncate'></p> */}
                <div className='flex items-center gap-1 my-1'>
                    <MapMarkerOutlineIcon className='w-4 h-4 text-orange' />
                    <p className='text-xs '>{event?.location}</p>
                </div>
 
                <OverlappingAvatars />
 
                <div className='flex w-full gap-2 my-3 overflow-x-scroll text-xs hideScrollbar scroll-smooth xmd:text-sm'>
                    {event?.tags.map((item: string, index: number) => {
                        return (
                            <div
                                key={index}
                                className='px-2 py-1 text-xs capitalize border rounded-md whitespace-nowrap orange-outline '
                            >
                                {item}
                            </div>
                        )
                    })}
                </div>
            </Link>
            {/* </Link> */}
        </div>
    )
}
