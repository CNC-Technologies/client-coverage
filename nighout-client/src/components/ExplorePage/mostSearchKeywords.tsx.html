import Link from 'next/link'

import { SearchOutlineIcon } from '@/constants/icons/icons'

export default function MostSearchKeywords() {
    const categories = [
        'Upcoming concerts',
        'Historical landmarks',
        'Beaches and resorts',
        'Local festivals',
        'Museums and galleries',
        'Outdoor adventures',
        'City tours',
        'Sports events',
        'Parks and gardens',
        'Nightlife and bars',
        'Art exhibitions',
        'Food and drink festivals',
    ]

    return (
        <div className='mt-12 smLaptop:mt-24'>
            <div>
                <h2 className='text-xl mobile:text-2xl xmd:text-[1.65rem] md:text-3xl desktop:text-4xl   font-medium'>
                    Most Searched Keywords
                </h2>
                <p className='mt-2 text-sm xmd:text-base smLaptop:text-lg text-secondary'>
                    Popular search keywords
                </p>
            </div>

            <div className='flex flex-wrap items-center gap-2 my-6 justify-left md:gap-4 '>
                {categories.map((category, index) => {
                    return (
                        <Link
                            href={`/search/?q=${category}`}
                            key={index}
                            className='min-w-fit md:min-w-[10rem] h-full'
                        >
                            <div className='border rounded-lg orange-outline px-2 py-1 flex gap-2 mobile:gap-5 justify-center items-center '>
                                <p className='max-md:text-sm font-medium'>
                                    {category}
                                </p>
                                <div className=' rounded-full p-1 flex justify-center items-center cursor-pointer'>
                                    <SearchOutlineIcon className='w-3 h-3  mobile:w-4 mobile:h-4' />
                                </div>
                            </div>
                        </Link>
                    )
                })}
            </div>
        </div>
    )
}
