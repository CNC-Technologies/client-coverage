import Image from 'next/image'
import Link from 'next/link'

import { MockedCategories } from '@/constants/MockData/CategoriesList'

export default function ExploreCategories({ categoryName }) {
    const explore = {
        heading: 'Explore categories on Placio.',
        subtext:
            'Find everything you need to know about the best restaurants, bars, clubs, and events in your area by browsing our categories.',
    }
    const category = {
        heading: `Explore different ${categoryName} categories`,
        subtext: `Your guide to the best ${categoryName} in town: Browse through the categories to discover a world of possibilities and plan your next adventure. `,
    }

    return (
        <div className='mt-5 smLaptop:mt-10'>
            <div>
                <h2 className='text-xl mobile:text-2xl md:text-3xl  text-primary  font-medium'>
                    {categoryName ? category.heading : explore.heading}
                </h2>
                <p className='text-sm xmd:text-base smLaptop:text-lg mt-2 w-full xmd:w-[90%] laptop:w-[70%] desktop:w-[50%] text-secondary'>
                    {categoryName ? category.subtext : explore.subtext}
                </p>
            </div>

            <div className='flex max-xxmd:overflow-x-scroll max-xxmd:hideScrollbar max-xxmd:scroll-smooth w-full xxmd:flex-wrap justify-left gap-4 desktop:gap-6 my-5'>
                {MockedCategories.map((category, index) => {
                    return (
                        <Link
                            href={`/search?q=${category?.name}`}
                            key={index}
                            className='max-xxmd:min-w-[12rem] w-[12rem] max-xxmd:h-[7rem] h-[5rem] md:h-[6rem] desktop:h-[7rem] rounded-xl relative'
                        >
                            <Image
                                src={category?.image}
                                alt={'category'}
                                priority={true}
                                quality={75}
                                width={100}
                                height={100}
                                className='w-full h-full object-cover rounded-xl object-center'
                            />
                            <div className='bg-primary/50 absolute top-0 left-0 w-full h-full rounded-lg flex justify-center items-center px-2'>
                                <p className='font-semibold text-white text-base smLaptop:text-xl text-center'>
                                    {category?.name}
                                </p>
                            </div>
                        </Link>
                    )
                })}
            </div>
        </div>
    )
}
