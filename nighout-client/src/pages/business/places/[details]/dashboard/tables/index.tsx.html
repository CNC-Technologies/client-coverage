import { useRouter } from 'next/router'
import { useState } from 'react'

import PageTransition from '@/components/Layout/PageTransition'
import PlacesSettingsLayout from '@/components/business/layouts/PlacesSettingsLayout'
// import dummyTables from '@/constants/business/dummyTables'
import Button from '@/components/FormComponents/button'
import NewTableForm from '@/components/forms/business/smart-menu/CreateTable'
import DialogLayout from '@/components/business/layouts/DialogLayout'
import {
    // mockTablesData,
    tableComponents,
} from '@/constants/business/dummyTables'
import { useGetAllTables } from '@/hooks/business/places/smart-menu/useTables'

export default function Tables() {
    const [newTable, setNewTable] = useState(false)
    const router = useRouter()
    const placeId = router?.query?.details as string

    const { data, isInitialLoading } = useGetAllTables(placeId)
    const tables = data?.data

    // console.log(tables)

    const renderTable = (table: any) => {
        const singleTable = tableComponents?.find(
            (item) => item?.id === table.capacity,
        )

        // Check if a table with the specified Table Capacity is found
        if (singleTable) {
            const TableComponent = singleTable.element

            // TableComponent
            return <TableComponent number={table.number || 0} />
        } else {
            const TableComponent = tableComponents[0].element
            return <TableComponent number={table.number || 0} />
        }

        // return null
    }

    const renderAvailability = (status) => {
        let colorClass, text

        switch (status) {
            case 'available':
                colorClass = 'bg-[green]'
                text = 'available'
                break
            case 'reserved':
                colorClass = 'bg-red-500'
                text = 'reserved'
                break
            case 'taken':
                colorClass = 'bg-orange'
                text = 'taken'
                break
            default:
                colorClass = 'bg-gray-500'
                text = 'Unknown'
        }

        return (
            <div
                className={`px-2 flex justify-center w-fit mx-auto rounded-lg ${colorClass}`}
            >
                <div className='text-white text-sm font-semibold capitalize'>
                    {text}
                </div>
            </div>
        )
    }

    const handleTableDetails = (tableId) => {
        // console.log(tableId)

        const currentPath = router.asPath

        router.push(`${currentPath}/${tableId}/details`)
    }

    return (
        <PageTransition>
            <PlacesSettingsLayout variant='placeDashboard'>
                <div className='row justify-end'>
                    <Button onClick={() => setNewTable(true)}>New Table</Button>
                    <DialogLayout
                        closeDialog={() => setNewTable(false)}
                        dialogOpened={newTable}
                        title={'New Table'}
                    >
                        <NewTableForm
                            handleCloseModal={() => setNewTable(false)}
                        />
                    </DialogLayout>
                </div>
                <div className='tables mt-5 pt-3 bg-[grey]/30 dark:bg-white/5 rounded-xl'>
                    <div className='row justify-end pr-3'>
                        <div className='bg-white text-sm px-3 rounded-xl flex items-center space-x-4 py-2'>
                            <div className='flex space-x-3 items-center'>
                                <div className='w-4 h-4 bg-[green] rounded-full'></div>
                                <div>Available</div>
                            </div>
                            <div className='flex space-x-3 items-center'>
                                <div className='w-4 h-4 bg-red-500 rounded-full'></div>
                                <div>Reserved</div>
                            </div>
                            <div className='flex space-x-3 items-center'>
                                <div className='w-4 h-4 bg-orange rounded-full'></div>
                                <div>Taken</div>
                            </div>
                        </div>
                    </div>

                    {isInitialLoading && !tables?.length ? (
                        <div className='row gap-8 flex-wrap  px-5 py-3 '>
                            {[...Array(12)].map((_, idx) => {
                                return (
                                    <div
                                        className='bg-[whitesmoke] rounded-full animate-pulse h-[130px] w-[130px]'
                                        key={idx}
                                    ></div>
                                )
                            })}
                        </div>
                    ) : (
                        <div className='row gap-8 flex-wrap  px-5 py-3 '>
                            {tables.map((table: any, idx) => (
                                <div
                                    key={idx}
                                    className='w-fit flex flex-col justify-center cursor-pointer rerelativelative '
                                >
                                    <div className=' transition-opacity duration-300 ease-in-out'>
                                        <div className='group relative'>
                                            {renderTable(table)}
                                            {/* Tooltip */}
                                            <div className='rounded-lg opacity-0 hidden min-w-[15rem] z-10 w-full group-hover:opacity-100 group-hover:block duration-300 ease-in-out bg-white p-2 shadow-md absolute -top-36 -left-32'>
                                                <div className='font-semibold text-center'>
                                                    T{table.number}
                                                </div>
                                                <div className='mt-2 grid gap-2'>
                                                    {/* <div className='flex justify-between space-x-5 text-xs'>
                                                        <div>
                                                            Tableâ€™s Waiter:
                                                        </div>
                                                        <div className='font-semibold'>
                                                            Tobi Ikupolati
                                                        </div>
                                                    </div> */}
                                                    {table.name && (
                                                        <div className='flex justify-between space-x-5 text-xs'>
                                                            <div>
                                                                Table name:
                                                            </div>
                                                            <div className='font-semibold'>
                                                                {table.name}
                                                            </div>
                                                        </div>
                                                    )}
                                                    {table.capacity && (
                                                        <div className='flex justify-between space-x-5 text-xs'>
                                                            <div>
                                                                Table Capacity:
                                                            </div>
                                                            <div className='font-semibold'>
                                                                {table.capacity}
                                                            </div>
                                                        </div>
                                                    )}
                                                    <div className='flex justify-between space-x-5 text-xs'>
                                                        <div>Table Type:</div>
                                                        <div className='font-semibold'>
                                                            {table.tableType}
                                                        </div>
                                                    </div>
                                                </div>
                                                <hr className='border-black/40 border-dashed my-2' />

                                                <div className='flex flex-col justify-center'>
                                                    <div className='text-xs text-center'>
                                                        Seat Availability
                                                    </div>
                                                    <div className='my-1 px-5 flex justify-center gap-2 items-center flex-wrap '>
                                                        {[
                                                            ...Array(
                                                                table.seats,
                                                            ),
                                                        ].map((_seat, idx) => {
                                                            const isOccupied =
                                                                idx < 3
                                                            // const isOccupied = idx < table.order.length;

                                                            return (
                                                                <div
                                                                    key={idx}
                                                                    className=''
                                                                >
                                                                    <div
                                                                        className={`w-3 h-3 rounded-full ${
                                                                            isOccupied
                                                                                ? 'bg-orange'
                                                                                : 'bg-[green]'
                                                                        }`}
                                                                    ></div>
                                                                </div>
                                                            )
                                                        })}
                                                    </div>
                                                </div>

                                                <div className='mt-1 flex justify-center'>
                                                    <Button
                                                        onClick={() =>
                                                            handleTableDetails(
                                                                table.id,
                                                            )
                                                        }
                                                        className='bg-orange hover:bg-orange hover:brightness-90'
                                                    >
                                                        Details
                                                    </Button>
                                                </div>
                                            </div>
                                        </div>
                                        <div>
                                            {renderAvailability(table.status)}
                                        </div>
                                    </div>

                                    {/* <div className='absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2 h-10 w-10 rounded-full bg-[green]/10 text-[green]'></div> */}
                                </div>
                            ))}
                        </div>
                    )}
                </div>
            </PlacesSettingsLayout>
        </PageTransition>
    )
}
