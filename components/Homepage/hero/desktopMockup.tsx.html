import Image from 'next/image'

import { useEffect, useRef, useState } from 'react'

import useImageCarousel from '@/hooks/useimageSlider'

import {
    desktopMockupImage1,
    desktopMockupImage2,
} from '@/constants/app/defaults'
import { DotSolidIcon } from '@/constants/icons/icons'
import useHandleColorChange from '@/hooks/useHandleColorChange'

const DesktopMockup = ({ setActiveClass }) => {
    // * The next 3 lines handle changing the page color when the component is in view.
    const idClass = 'theme-purple'
    const containerRef = useRef(null)
    useHandleColorChange(containerRef, idClass, setActiveClass)

    const images = [
        { src: desktopMockupImage1, text: 'Connect' },
        { src: desktopMockupImage2, text: 'Explore' },
    ]

    const [activeIndex] = useImageCarousel(
        images.map((image) => image.src),
        8000,
    )

    const activeElement = images[activeIndex as number]

    const [innerHeight, setInnerHeight] = useState(0)
    useEffect(() => {
        if (typeof window !== 'undefined') {
            setInnerHeight(window.innerHeight)
            window.addEventListener('resize', () =>
                setInnerHeight(window.innerHeight),
            )

            if (typeof window !== 'undefined')
                return () => {
                    window.removeEventListener('resize', () =>
                        setInnerHeight(window.innerHeight),
                    )
                }
        }
    }, [])

    const setElementMargin = (innerHeight: number) => {
        if (innerHeight < 480) {
            return '-mt-[8%]'
        } else if (innerHeight < 730) {
            return '-mt-[20%]'
        } else {
            return 'max-mobile:-mt-[75%] -mt-[40%] xmd:-mt-[25%]'
        }
    }

    return (
        <div
            ref={containerRef}
            id='desktopMockUp'
            className={`lgDesktop:px-14 w-full  max-width z-20 mx-auto px-6 mobile:px-10 md:px-14 relative ${setElementMargin(
                innerHeight,
            )}  md:-mt-[10%] mb-2`}
        >
            <div className='w-full h-full'>
                <div className='flex items-center h-8 gap-1 px-2 rounded-tl-lg rounded-tr-lg bg-grey'>
                    {[...Array(3)].map((_, index) => {
                        return (
                            <div key={index}>
                                <DotSolidIcon
                                    className={`${
                                        index === 0
                                            ? 'text-[#F3605C]'
                                            : index === 1
                                            ? 'text-[#F8BE39]'
                                            : 'text-[#50C845]'
                                    }`}
                                />
                            </div>
                        )
                    })}
                </div>

                <div className='h-[50rem] max-laptop:h-[34rem] max-xmd:h-[25rem] max-xxmobile:h-[20rem] relative bg-white  rounded-bl-lg rounded-br-lg overflow-hidden'>
                    <div className='px-6 xmd:px-10 laptop:px-16'>
                        <Image
                            height={100}
                            width={100}
                            src={`/Images/whiteLogo.svg`}
                            alt='Placio Logo'
                            className='w-40 h-20 max-laptop:w-32 max-laptop:h-20 max-md:w-28 max-xmd:h-12 max-xmd:w-16'
                        />
                    </div>

                    <div className='mx-auto mb-8 text-center text-primary'>
                        <h1 className=' tracking-normal w-[90%] laptop:w-[50%] mx-auto text-6xl max-laptop:text-[3.3rem] max-md:text-[2.5rem] max-xmd:text-[2rem]  max-lgMobile:leading-tight  max-mobile:text-3xl max-xmobile:text-2xl font-medium my-6 max-laptop:my-2'>
                            One platform to{' '}
                            <span
                                className={`font-semibold transition duration-1000 ease-in-out`}
                            >
                                {activeElement.text}
                            </span>
                        </h1>

                        <p className='text-lg w-[90%] laptop:w-[60%] mx-auto my-5 max-xmobile:my-2 max-md:text-base max-mobile:text-sm max-xmobile:text-xs'>
                            Lorem ipsum dolor sit amet consectetur. Neque enim
                            aliquam tincidunt duis phasellus. Elit amet
                            vulputate ut libero id tempor convallis turpis.
                        </p>
                    </div>

                    <div className='h-[32rem] max-laptop:h-[25rem] max-xmd:h-[12rem] max-xmobile:h-[8rem] max-xxmobile:h-[6.5rem] extraLarge:h-[45rem] relative mt-12 max-laptop:mt-0 extraLarge:mt-10'>
                        <Image
                            src={activeElement.src}
                            alt={'Desktop mockup'}
                            priority={true}
                            quality={75}
                            width={100}
                            height={100}
                            className='object-contain object-center w-full h-full transition-opacity duration-1000 rounded-bl-lg rounded-br-lg opacity-100 '
                        />
                        {images.map((image, index) => (
                            <div
                                key={index}
                                className={`h-full w-full absolute bottom-0 left-0 bg-center bg-cover bg-no-repeat  transition-opacity duration-1000 rounded-bl-lg rounded-br-lg  ${
                                    activeIndex !== index
                                        ? 'opacity-0'
                                        : 'opacity-100'
                                }`}
                            >
                                <Image
                                    src={image.src}
                                    alt={'Desktop mockup'}
                                    priority={true}
                                    quality={75}
                                    width={100}
                                    height={100}
                                    className='object-contain object-center w-full h-full rounded-bl-lg rounded-br-lg '
                                />
                            </div>
                        ))}
                    </div>
                </div>
            </div>
        </div>
    )
}

export default DesktopMockup
