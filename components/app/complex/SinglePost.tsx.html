import Link from 'next/link'
import { useState } from 'react'

import Avatar from '@/components/app/simple/Avatar'

import { IconButton } from '@/components/FormComponents/button'
import ShareButton from '@/components/app/simple/ShareButton'
import LikeItem from '@/components/app/simple/LIkeItem'

import {
    ChatOutlineIcon,
    ForwardOutlineIcon,
    // ThreeDotsIcon,
} from '@/constants/icons/icons'
import ParentModal from '@/components/modals/parentModal'
import CommentItem from '@/components/app/compound/Post/CommentItem'
import CommentField from '@/components/app/compound/Post/CommentField'
import { Post } from '@/interfaces/posts.interface'
import { formatMonthAndDate } from '@/utils/dateFormat'
import PostMedias from '@/components/app/compound/Post/PostMedias'

export default function SinglePost({ post }: { post: Post | null }) {
    const [commentsModal, setCommentsModal] = useState(false)

    return (
        <div className='pb-3 space-y-3 border-b col border-grey'>
            {/* Post creator profile*/}
            <div className='items-center space-x-3 row'>
                <Link href={`/${post?.username}`}>
                    <Avatar img={post?.profilePicture} size='md' />
                </Link>
                <div className='flex-1 col'>
                    <div className='flex items-center gap-2'>
                        <Link
                            href={`/${post?.username}`}
                            className='font-semibold'
                        >
                            {post?.username}
                        </Link>
                        <p className='text-sm'>
                            {formatMonthAndDate(post?.createdAt as string)}
                        </p>
                    </div>
                    <div className='text-xs text-secondary'>Abuja</div>
                </div>
                {/* <IconButton className='primary-outline'>
                    <ThreeDotsIcon />
                </IconButton> */}
            </div>

            {/* Post content */}
            <div>
                <p className='text-sm md:text-base text-primary text-normal leading-[18px]'>
                    {post?.content}
                </p>
            </div>

            {/* Post images and videos */}
            {post?.medias?.length !== 0 && (
                <PostMedias
                    username={post?.username as string}
                    medias={post?.medias}
                />
            )}

            {/* Second row */}
            <div className='items-center justify-between row'>
                <div className='items-center space-x-3 row text-primary'>
                    <div className='flex items-center space-x-1 '>
                        <LikeItem background={true} />
                        <p className='text-sm font-medium'>
                            {post?.likes?.length}
                        </p>
                    </div>
                    <div className='flex items-center space-x-1'>
                        <IconButton onClick={() => setCommentsModal(true)}>
                            <ChatOutlineIcon className='w-5 h-5' />
                        </IconButton>
                        <p className='text-sm font-medium'>
                            {post?.comments?.length}
                        </p>
                    </div>
                </div>

                <ShareButton
                    content={{
                        url: 'https://placio.io',
                        text: post?.content as string,
                        // media: post?.medias[0]?.url,
                    }}
                >
                    <button className='flex items-center h-10 gap-1 px-4 text-center rounded-3xl primary-outline'>
                        Share
                        <ForwardOutlineIcon />
                    </button>
                </ShareButton>
            </div>

            {/* Comment Section */}
            <CommentField />

            {/* Comments Modal  */}
            <ParentModal
                isOpen={commentsModal}
                handleClose={() => setCommentsModal(false)}
                extraClasses={undefined}
            >
                <div className='relative'>
                    <h4 className='text-center px-2 text-lg font-semibold sticky top-0 bg-white'>
                        Comments
                    </h4>

                    <div className='mt-4 grid gap-2'>
                        {post?.comments?.map((comment, idx) => {
                            return <CommentItem key={idx} comment={comment} />
                        })}
                    </div>

                    <div className='sticky border-t  bg-white py-2 bottom-0 left-0 w-full'>
                        <CommentField />
                    </div>
                </div>
            </ParentModal>
        </div>
    )
}
