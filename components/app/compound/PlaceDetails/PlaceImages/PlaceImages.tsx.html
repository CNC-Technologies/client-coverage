import { useRouter } from 'next/router'
import { useEffect, useState } from 'react'

import dynamic from 'next/dynamic'

import Button from '@/components/FormComponents/button'
import AllImagesLayout from '@/components/app/compound/PlaceDetails/PlaceImages/AllImagesLayout'
import ImagesModal from '@/components/app/compound/PlaceDetails/PlaceImages/ImagesModal'
import { samplePlaceImages } from '@/constants/MockData/PlaceDetails/PlaceDetailsImages'
import { samplePlace } from '@/constants/app/defaults'
import {
    CameraOutlineIcon,
    ChevronLeftIcon,
    ChevronRightIcon,
    GridSolidIcon,
} from '@/constants/icons/icons'

const Image = dynamic(import('next/image'))

export default function PlaceDetailsImages() {
    const [selectedImageIndex, setSelectedImageIndex] = useState(0)
    const [isOpen, setIsOpen] = useState(false)

    const router = useRouter()
    const pathname = router.asPath
    const { media } = router.query
    const placeName = router.query.details

    const closeDialog = () => {
        const newPath = pathname.split('?')[0]
        return newPath
    }

    useEffect(() => {
        if (media === 'true') {
            setIsOpen(true)
        } else {
            setIsOpen(false)
        }
    }, [media, pathname])

    const handlePreviousClick = () => {
        setSelectedImageIndex((prevIndex) =>
            prevIndex === 0 ? samplePlaceImages.length - 1 : prevIndex - 1,
        )
    }

    const handleNextClick = () => {
        setSelectedImageIndex((prevIndex) =>
            prevIndex === samplePlaceImages.length - 1 ? 0 : prevIndex + 1,
        )
    }

    return (
        <div>
            <div className='grid grid-cols-4 grid-rows-4 grid-flow-col gap-2 h-[350px] xlaptop:h-[400px] desktop:h-[450px]'>
                <div className='relative w-full h-full col-span-4 row-span-4 xlaptop:col-span-3'>
                    <div className='relative w-full h-full '>
                        <button
                            className='absolute z-10 transform -translate-y-1/2 rounded-full shadow-sm top-1/2 left-3 white-fill dark:bg-grey'
                            onClick={handlePreviousClick}
                        >
                            <ChevronLeftIcon className='w-8 h-8' />
                        </button>
                        <button
                            className='absolute z-10 transform -translate-y-1/2 rounded-full shadow-sm top-1/2 right-3 white-fill dark:bg-grey'
                            onClick={handleNextClick}
                        >
                            <ChevronRightIcon className='w-8 h-8' />
                        </button>
                        <div className='absolute top-0 bottom-0 left-0 right-0 w-full h-full'>
                            {samplePlaceImages.map((image, index) => (
                                <Image
                                    key={index}
                                    className={`w-full h-full object-cover object-center  ${
                                        index === selectedImageIndex
                                            ? ''
                                            : 'hidden'
                                    }`}
                                    src={image}
                                    alt={'place name'}
                                    priority={true}
                                    quality={75}
                                    width={100}
                                    height={100}
                                />
                            ))}

                            <div className='absolute bottom-3 right-2'>
                                <Button
                                    onClick={() =>
                                        router.push(`${pathname}?media=true`)
                                    }
                                    className='white-fill flex space-x-2'
                                >
                                    <p>See all photos</p>
                                    <GridSolidIcon className='text-xl' />
                                </Button>
                            </div>
                        </div>
                    </div>
                </div>

                {[...Array(2)].map((_, index) => {
                    return (
                        <div
                            key={index}
                            className='relative hidden col-span-1 row-span-2 cursor-pointer xlaptop:block hover:brightness-150 smoothen'
                        >
                            <Image
                                src={samplePlace}
                                alt={'category'}
                                priority={true}
                                quality={75}
                                width={100}
                                height={100}
                                className='object-cover object-center w-full h-full '
                            />
                            <div className='absolute top-0 left-0 z-10 flex flex-col items-center justify-center w-full h-full gap-2 text-white dark-overlay'>
                                <p>Category {index}</p>
                                <CameraOutlineIcon className='text-2xl' />
                            </div>
                        </div>
                    )
                })}
            </div>

            {/* Thumbnails  */}
            <div className='grid grid-col-1 mt-5'>
                <div className='w-full overflow-x-scroll row xlaptop:mt-2 scroll-smooth'>
                    {samplePlaceImages.map((image, index) => (
                        <div className='col' key={index}>
                            <div
                                className={`relative h-24 w-28 rounded-lg border-2 border-gray-300 ${
                                    index === selectedImageIndex
                                        ? 'border-blue-500'
                                        : ''
                                }`}
                                onClick={() => setSelectedImageIndex(index)}
                            >
                                {index !== selectedImageIndex && (
                                    <div className='absolute top-0 left-0 z-10 w-full h-full dark-overlay'></div>
                                )}
                                <Image
                                    className='object-cover w-full h-full'
                                    src={image}
                                    alt='category'
                                    priority={true}
                                    quality={75}
                                    width={100}
                                    height={100}
                                />
                            </div>
                        </div>
                    ))}
                </div>
            </div>

            {/* See all Images Modal  */}
            <ImagesModal
                isOpen={isOpen}
                handleClose={() => router.push(closeDialog())}
                extraClasses={undefined}
            >
                <AllImagesLayout placeName={placeName} />
            </ImagesModal>
        </div>
    )
}
